const pool = new Pool({ max: 20, idleTimeoutMillis: 30000, connectionTimeoutMillis: 2000
}); app.get('/users/:id', async (req, res) => { const client = await pool.connect(); try { const result = await client.query('SELECT * FROM users WHERE id = $1', [req.params.id]); return res.json(result.rows[0]); } finally { client.release(); }
});
