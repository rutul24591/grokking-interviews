<!-- SearchBar.jsx - React Island -->
import { useState } from 'react';

export default function SearchBar() {
  const [query, setQuery] = useState('');

  return (
    <input
      type="search"
      value={query}
      onChange={(e) => setQuery(e.target.value)}
      placeholder="Search..."
    />
  );
}

// This React component only hydrates when client:load directive is used
// Without the directive, it renders static HTML only
