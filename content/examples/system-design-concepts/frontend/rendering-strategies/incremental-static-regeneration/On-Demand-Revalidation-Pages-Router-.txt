// pages/api/revalidate.ts
import type { NextApiRequest, NextApiResponse } from 'next';

export default async function handler(
  req: NextApiRequest,
  res: NextApiResponse
) {
  // Verify webhook secret
  if (req.query.secret !== process.env.REVALIDATE_SECRET) {
    return res.status(401).json({ message: 'Invalid token' });
  }

  try {
    const { productId } = req.body;

    // Revalidate specific product page
    await res.revalidate(`/products/\${productId}`);

    // Also revalidate related pages
    await res.revalidate('/products');
    await res.revalidate('/');

    return res.json({
      revalidated: true,
      timestamp: new Date().toISOString()
    });
  } catch (err) {
    return res.status(500).json({ message: 'Error revalidating', error: err });
  }
}

// Webhook from CMS (Contentful, Sanity, etc.):
// POST https://yoursite.com/api/revalidate?secret=YOUR_SECRET
// Body: { "productId": "123" }
